98 std::vector<uint8_t> compress_chunk(
99 const std::vector<uint8_t>& chunk,
100 CompressMethod method,
101 uint32_t win_size) const
102 {
103 switch (method) {
104 case CompressMethod::HUFFMAN: {
105 HuffmanCoder hc;
106 return hc.encode(chunk);
107 }
108 case CompressMethod::LZ77: {
109 LZ77Coder lz(win_size);
110 auto lz_out = lz.encode(chunk);
111 HuffmanCoder hc;
112 return hc.encode(lz_out);
113 }
114 case CompressMethod::PPM_LZSS_RNC: {
115 LZSSCoder lzss;
116 auto lzss_out = lzss.encode(chunk);
117 RNCCoder rnc;
118 return rnc.encode(lzss_out);
119 }
120 }
121 return chunk;
122 }

378 void decompress(const std::string& input_path,
379 const std::string& output_path)
380 {
381 cancelled_ = false;
382
383 status("Открытие архива...");
384 progress(0.0);
385
386 ArchiveHeader hdr;
387 std::vector<uint8_t> payload;
388 try {
389 auto [h, d] = read_archive(input_path);
390 hdr = std::move(h);
391 payload = std::move(d);
392 } catch (const std::runtime_error& e) {
393 throw std::runtime_error(
394 std::string("Повреждённый архив: ") + e.what());
395 }
396
397 status("Формат: " + format_name(hdr.format) +
398 ", Метод: " + method_name(hdr.method));
399 progress(0.05);
400
401 if (payload.size() < 4)
402 throw std::runtime_error("Архив слишком короткий");
403
404 size_t pos = 0;
405 auto rb4 = [&]() -> uint32_t {
406 if (pos + 4 > payload.size())
407 throw std::runtime_error("Неожиданный конец данных архива");
408 uint32_t v = (uint32_t(payload[pos])<<24) | (uint32_t(payload[pos+1])<<16)
409 | (uint32_t(payload[pos+2])<<8) | payload[pos+3];
410 pos += 4;
411 return v;
412 };
413
414 uint32_t n_chunks = rb4();
415 if (n_chunks == 0 || n_chunks > 1024*1024)
416 throw std::runtime_error("Некорректное количество чанков: " +
417 std::to_string(n_chunks));
418
419 std::ofstream fout(output_path, std::ios::binary);
420 if (!fout) throw std::runtime_error("Не удалось создать файл: " + output_path);
421
422 uint32_t crc_accum = 0xFFFFFFFFu;
423 uint64_t total_written = 0;
424
425 for (uint32_t i = 0; i < n_chunks; ++i) {
426 if (cancelled_) throw std::runtime_error("Отменено пользователем");
427
428 double p = 0.05 + 0.90 * static_cast<double>(i) / n_chunks;
429 progress(p);
430 char pct[8]; snprintf(pct, sizeof(pct), "%.0f%%", p * 100.0);
431 status(std::string("Распаковка... ") + pct);
432
433 uint32_t csz = rb4();
434 uint32_t orig_sz = rb4();
435
436 bool raw = (orig_sz & 0x80000000u) != 0;
437 orig_sz &= 0x7FFFFFFFu;
438 uint32_t data_sz = raw ? orig_sz : csz;
439
440 if (pos + data_sz > payload.size())
441 throw std::runtime_error("Усечённый чанк " + std::to_string(i));
442
443 std::vector<uint8_t> cdata(payload.begin() + static_cast<ptrdiff_t>(pos),
444 payload.begin() + static_cast<ptrdiff_t>(pos) + data_sz);
445 pos += data_sz;
446
447 std::vector<uint8_t> result;
448 if (raw) {
449 result = std::move(cdata);
450 } else {
451 try {
452 result = decompress_chunk(cdata, hdr.method, hdr.win_size);
453 } catch (const std::exception& e) {
454 throw std::runtime_error(
455 std::string("Ошибка распаковки чанка ") +
456 std::to_string(i) + ": " + e.what());
457 }
458 }
459
460 if (result.size() != orig_sz)
461 throw std::runtime_error(
462 "Размер чанка " + std::to_string(i) + " не совпадает");
463
464 crc_accum = CRC32::update(crc_accum, result.data(), result.size());
465 fout.write(reinterpret_cast<const char*>(result.data()),
466 static_cast<std::streamsize>(result.size()));
467 total_written += result.size();
468 }
469 fout.close();
470
471 if (total_written != hdr.orig_size)
472 throw std::runtime_error(
473 "Итоговый размер (" + human_size(total_written) +
474 ") не совпадает с ожидаемым (" + human_size(hdr.orig_size) + ")");
475
476 status("Проверка CRC32...");
477 progress(0.97);
478 uint32_t calc_crc = CRC32::finalize(crc_accum);
479 if (calc_crc != hdr.crc32)
480 throw std::runtime_error(
481 "CRC32 не совпадает — архив повреждён. "
482 "Ожидалось: " + crc_hex(hdr.crc32) +
483 ", получено: " + crc_hex(calc_crc));
484
485 remove_checkpoint(input_path);
486 status("Восстановлено: " + human_size(total_written) +
487 " → \"" + hdr.filename + "\"");
488 progress(1.0);
