158inline ArchiveOffsets begin_stream_archive(std::fstream& f,
159 const ArchiveHeader& hdr)
160{
161 const char* magic = (hdr.format == ArchiveFormat::CZIP) ? "CZIP" : "CRAR";
162 f.write(magic, 4);
163 write_u8(f, 0x01);
164 write_u8(f, static_cast<uint8_t>(hdr.method));
165 write_u8(f, hdr.flags);
166 write_u32(f, hdr.win_size);
167
168 ArchiveOffsets off;
169 off.orig_size_pos = f.tellp();
170 write_u64(f, 0);
171
172 off.crc32_pos = f.tellp();
173 write_u32(f, 0);
174
175 write_u16(f, static_cast<uint16_t>(hdr.filename.size()));
176 f.write(hdr.filename.data(), static_cast<std::streamsize>(hdr.filename.size()));
177
178 off.n_chunks_pos = f.tellp();
179 write_u32(f, 0);
180
181 return off;
182}

249 ArchiveHeader hdr;
250 if (std::memcmp(magic, "CZIP", 4) == 0) hdr.format = ArchiveFormat::CZIP;
251 else if (std::memcmp(magic, "CRAR", 4) == 0) hdr.format = ArchiveFormat::CRAR;
252 else throw std::runtime_error(

256 if (ver != 0x01) throw std::runtime_error("Неподдерживаемая версия архива");

258 hdr.method = static_cast<CompressMethod>(read_u8(f));
259 hdr.flags = read_u8(f);

261 hdr.orig_size = read_u64(f);
262 hdr.crc32 = read_u32(f);
