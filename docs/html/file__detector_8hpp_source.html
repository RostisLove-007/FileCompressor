88 static bool looks_like_text(const std::vector<uint8_t>& data) {
89 size_t check = std::min(data.size(), size_t(512));
90 size_t non_text = 0;
91 for (size_t i = 0; i < check; ++i) {
92 uint8_t b = data[i];
93 if (b == 0) return false;
94 if (b < 0x09 && b != 0x00) ++non_text;
95 else if (b > 0x0D && b < 0x20) ++non_text;
96 }
97 return non_text < check / 20;
98 }
